#include <iostream>

#include "TSystem.h"
#include "TSysEvtHandler.h"

class TInterrupt: public TSignalHandler {

 public:
  Int_t NumSig;
  static Int_t fgInterrupted;

  Bool_t IsInterrupted(){ return( fgInterrupted>0);};
  Bool_t IsNotInterrupted(){ return( fgInterrupted==0);};

};
